6   package edu.jhu.hlt.concrete.util;
7   
8   import edu.jhu.hlt.concrete.Communication;
9   import edu.jhu.hlt.concrete.TextSpan;
10  
11  /**
12   * A wrapper around {@link TextSpan} offering improved functionality. 
13   * 
14   * @author max
15   */
16  public class SuperTextSpan {
17  
18    private final TextSpan ts;
19    private final Communication c;
20    
21    /**
22     * Create a {@link SuperTextSpan} object.
23     * The {@link Communication} object should be the one that is associated
24     * with the TextSpan that was passed in.
25     *  
26     * @param ts - a {@link TextSpan} object
27     * @param c - a {@link Communication} object from which the {@link TextSpan} object was generated
28     */
29    public SuperTextSpan(TextSpan ts, Communication c) {
30      this.ts = ts;
31      this.c = c;
32    }
33  
34    /**
35     * Get the text from a {@link TextSpan}. May throw an {@link IndexOutOfBoundsException}
36     * if the {@link Communication} and {@link TextSpan} do not align properly.
37     * 
38     * @return a {@link String} with the text of the {@link TextSpan} object. 
39     */
40    public String getText() {
41      return c
42          .getText()
43          .substring(ts.getStart(), ts.getEnding());
44    }
45  }
