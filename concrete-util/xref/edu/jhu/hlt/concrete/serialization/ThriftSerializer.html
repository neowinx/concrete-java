5   package edu.jhu.hlt.concrete.serialization;
6   
7   import java.io.InputStream;
8   import java.nio.file.Path;
9   
10  import org.apache.thrift.TBase;
11  import org.apache.thrift.TFieldIdEnum;
12  
13  import edu.jhu.hlt.concrete.util.ConcreteException;
14  
15  /**
16   * Interface for de/serialization of 'thrift-like' objects. These objects
17   * are Java classes that are produced from Apache Thrift.
18   */
19  public interface ThriftSerializer<T extends TBase<T, ? extends TFieldIdEnum>> {
20    public T fromBytes(T base, byte[] bytes) throws ConcreteException;
21    public byte[] toBytes(T base) throws ConcreteException;
22    public T fromPath(T base, Path path) throws ConcreteException;
23    public T fromPathString(T base, String pathString) throws ConcreteException;
24    public T fromInputStream(T base, InputStream is) throws ConcreteException;
25  }
