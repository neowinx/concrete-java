6   package edu.jhu.hlt.concrete.validation;
7   
8   import java.util.Iterator;
9   
10  import edu.jhu.hlt.concrete.Communication;
11  import edu.jhu.hlt.concrete.EntityMention;
12  import edu.jhu.hlt.concrete.EntityMentionSet;
13  import edu.jhu.hlt.concrete.UUID;
14  
15  /**
16   * @author max
17   *
18   */
19  public class ValidatableEntityMentionSet extends AbstractAnnotation<EntityMentionSet> {
20  
21    /**
22     * @param annotation
23     */
24    public ValidatableEntityMentionSet(EntityMentionSet annotation) {
25      super(annotation);
26    }
27  
28    /* (non-Javadoc)
29     * @see edu.jhu.hlt.concrete.validation.AbstractAnnotation#isValidWithComm(edu.jhu.hlt.concrete.Communication)
30     */
31    @Override
32    protected boolean isValidWithComm(Communication c) {
33      boolean validMens = true;
34      Iterator<EntityMention> menIter = this.annotation.getMentionListIterator();
35      while (validMens && menIter.hasNext()) {
36        EntityMention em = menIter.next();
37        validMens = this.printStatus("EntityMention: " + em.getUuid().toString() + " must be valid", new ValidatableEntityMention(em).validate(c));
38      }
39      
40      return validMens;
41    }
42  
43    /* (non-Javadoc)
44     * @see edu.jhu.hlt.concrete.validation.AbstractAnnotation#isValid()
45     */
46    @Override
47    public boolean isValid() {
48      boolean init = this.validateUUID(this.annotation.getUuid())
49          && this.printStatus("Metadata must be valid", new ValidatableMetadata(this.annotation.getMetadata()).isValid())
50          && this.printStatus("EntityMentionSet must be set.", this.annotation.isSetMentionList());
51      if (init) {
52        boolean validMens = true;
53        Iterator<EntityMention> menIter = this.annotation.getMentionListIterator();
54        while (validMens && menIter.hasNext()) {
55          EntityMention em = menIter.next();
56          UUID emId = em.getUuid();
57          validMens = this.printStatus("EntityMention: " + emId.toString() + " must be valid.", new ValidatableEntityMention(em).isValid());
58        }
59        
60        return validMens;
61      } else
62        return false;
63    }
64  
65  }
