6   package edu.jhu.hlt.concrete.random;
7   
8   import java.util.HashSet;
9   import java.util.Random;
10  import java.util.Set;
11  
12  import edu.jhu.hlt.concrete.Communication;
13  import edu.jhu.hlt.concrete.metadata.AnnotationMetadataFactory;
14  import edu.jhu.hlt.concrete.uuid.UUIDFactory;
15  
16  /**
17   * Class that allows for construction of random (in the sense of randomness)
18   * Concrete objects.
19   */
20  public class RandomConcreteFactory {
21  
22    private final Random r;
23    private static final String[] COMM_TYPES = new String[] { "Document", "Tweet", "Other" };
24    private static final int COMM_TYPE_SIZE = COMM_TYPES.length;
25  
26    public RandomConcreteFactory() {
27      this.r = new Random();
28    }
29  
30    public RandomConcreteFactory(long seed) {
31      this.r = new Random(seed);
32    }
33  
34    public final String communicationType() {
35      return COMM_TYPES[this.r.nextInt(COMM_TYPE_SIZE)];
36    }
37  
38    public Communication communication() {
39      return new Communication()
40        .setUuid(UUIDFactory.newUUID())
41        .setId("corpus_" + this.r.nextInt(Integer.MAX_VALUE))
42        .setText("Some sample text.")
43        .setType(this.communicationType())
44        .setMetadata(AnnotationMetadataFactory.fromCurrentLocalTime()
45            .setTool("RandomConcreteFactory"));
46    }
47  
48    public Set<Communication> communicationSet(int nMembers) {
49      Set<Communication> cSet = new HashSet<>(nMembers + 1);
50      for (int i = 0; i < nMembers; i++)
51        cSet.add(this.communication());
52      // could get some dupes
53      while (cSet.size() < nMembers)
54        cSet.add(this.communication());
55      return cSet;
56    }
57  }
