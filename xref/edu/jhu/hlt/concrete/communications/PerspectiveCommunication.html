6   package edu.jhu.hlt.concrete.communications;
7   
8   import edu.jhu.hlt.concrete.AnnotationMetadata;
9   import edu.jhu.hlt.concrete.Communication;
10  import edu.jhu.hlt.concrete.Section;
11  import edu.jhu.hlt.concrete.Sentence;
12  import edu.jhu.hlt.concrete.TheoryDependencies;
13  import edu.jhu.hlt.concrete.util.ConcreteException;
14  
15  /**
16   * Attempt to prepare a new "perspective" from a {@link Communication}.
17   */
18  public class PerspectiveCommunication {
19  
20    private final Communication comm;
21  
22    /**
23     * @throws ConcreteException
24     *
25     */
26    public PerspectiveCommunication(Communication original, String perspectiveTool) throws ConcreteException {
27      if (!original.isSetText())
28        throw new ConcreteException("The original communication has to have .text set in order to build a perspective.");
29      Communication copy = new Communication(original);
30  
31      // Unset .text, then move it to originalText field.
32      copy.unsetText();
33      copy.setOriginalText(original.getText());
34  
35      // Move spans. Unset tokenizations.
36      if (copy.isSetSectionList())
37        for (Section s : copy.getSectionList()) {
38          s.setRawTextSpan(s.getTextSpan());
39          s.unsetTextSpan();
40  
41          if (s.isSetSentenceList())
42            for (Sentence ss : s.getSentenceList()) {
43              ss.setRawTextSpan(ss.getTextSpan());
44              ss.unsetTextSpan();
45  
46              if (ss.isSetTokenization())
47                ss.unsetTokenization();
48            }
49        }
50  
51      // Handle metadata.
52      AnnotationMetadata mdcp;
53      if (copy.isSetMetadata())
54        mdcp = new AnnotationMetadata(copy.getMetadata());
55      else
56        mdcp = new AnnotationMetadata();
57  
58      if (!mdcp.isSetDependencies())
59        mdcp.setDependencies(new TheoryDependencies());
60      mdcp.getDependencies().addToCommunicationsList(original.getUuid());
61      copy.setMetadata(mdcp);
62  
63      this.comm = copy;
64    }
65  
66    /**
67     * Return a {@link Communication} that represents this {@link PerspectiveCommunication}.
68     *
69     * It is not a pointer to the original, so changes to this communication will not propagate
70     * back to this object.
71     *
72     * @return a new {@link Communication}, not a pointer, to this {@link PerspectiveCommunication}.
73     */
74    public Communication getPerspective() {
75      return new Communication(this.comm);
76    }
77  }
