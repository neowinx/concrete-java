6   package edu.jhu.hlt.concrete.validation;
7   
8   import edu.jhu.hlt.concrete.Communication;
9   import edu.jhu.hlt.concrete.Section;
10  
11  /**
12   * @author max
13   *
14   */
15  public class ValidatableSection extends AbstractAnnotation<Section> {
16  
17    /**
18     * @param annotation
19     */
20    public ValidatableSection(Section annotation) {
21      super(annotation);
22    }
23  
24    @Override
25    protected boolean isValidWithComm(Communication c) {
26      // No text --> false
27      if (!this.printStatus("Text must be set in the comm", c.isSetText())) return false;
28      // For Sections: need to ensure that this TextSpan
29      // is valid in the context of the communication.
30      return this.printStatus("TextSpan must be valid wrt comm", new ValidatableTextSpan(this.annotation.getTextSpan()).isValidWithComm(c));
31    }
32  
33    @Override
34    public boolean isValid() {
35      return
36          this.validateUUID(this.annotation.getUuid())
37          // Hard-coded to text validity.
38          // TODO: update to Audio if ever is used.
39          && this.printStatus("SectionKind must be set", this.annotation.isSetKind())
40          && this.printStatus("TextSpan must be set", this.annotation.isSetTextSpan())
41          && this.printStatus("TextSpan must be valid", new ValidatableTextSpan(this.annotation.getTextSpan()).isValid());
42    }
43  }
