6   package edu.jhu.hlt.concrete.section;
7   
8   import edu.jhu.hlt.concrete.Communication;
9   import edu.jhu.hlt.concrete.Section;
10  import edu.jhu.hlt.concrete.TextSpan;
11  import edu.jhu.hlt.concrete.uuid.UUIDFactory;
12  
13  /**
14   * A class that exposes a method for converting {@link String} objects,
15   * representing contents of a {@link Communication} for example, into a single
16   * Concrete {@link Section}.
17   */
18  public class SingleSectionSegmenter {
19  
20    private SingleSectionSegmenter() {
21  
22    }
23  
24    /**
25     * Create a single {@link Section} based on some text, with the kind set to
26     * sectionKind.
27     *
28     * @param text the {@link String} upon which to create the Section object
29     * @param sectionKind the kind of the Section
30     * @return a {@link Section} with one {@link TextSpan}.
31     */
32    public static final Section createSingleSection (String text, String sectionKind) {
33      TextSpan ts = new TextSpan(0, text.length());
34  
35      Section s = new Section();
36      s.setUuid(UUIDFactory.newUUID());
37      s.setKind(sectionKind);
38      s.setTextSpan(ts);
39  
40      return s;
41    }
42  }
