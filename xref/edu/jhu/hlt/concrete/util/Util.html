25    /**
26     * Determine if a string is truly a {@link UUID} string.
27     *
28     * Unfortunately, an exception will be thrown if it is not valid.
29     *
30     * @param uuidStr - {@link String} to check
31     * @return true if a {@link UUID} string, otherwise false.
32     */
33    public static boolean isValidUUIDString(String uuidStr) {
34      // Do what we can to avoid an exception.
35      if (uuidStr != null && uuidStr.length() == 36 && uuidStr.contains("-"))
36        // Could do more above, but just eat the exception
37        // if it's invalid
38        try {
39          UUID.fromString(uuidStr);
40          return true;
41        } catch (IllegalArgumentException iae) {
42          return false;
43        }
44      else
45        return false;
46    }
