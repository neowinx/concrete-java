67    /**
68     * Single arg ctor: pass in a {@link Communication} object to wrap.
69     */
70    public SuperCommunication(Communication comm) {
71      // Create a copy, providing some immutability.
72      this.comm = new Communication(comm);
73      this.ser = new CompactCommunicationSerializer();
74    }

76    /**
77     * Return a {@link Section} that encompasses the entire {@link Communication} .text
78     * field.
79     *
80     * @param sectionKind the kind of section to generate
81     * @return a {@link Section} with the appropriate kind that spans the entire text
82     * @throws AnnotationException if the .text field is not set in this {@link Communication} object.
83     */
84    public Section singleSection(String sectionKind) throws AnnotationException {
85      if (!this.comm.isSetText())
86        throw new AnnotationException("This method requires the .text field to be set.");
87  
88      Section s = new Section(UUIDFactory.newUUID(), sectionKind);
89      TextSpan ts = new TextSpan(0, this.comm.getText().length());
90      s.setTextSpan(ts);
91  
92      return s;
93    }

126   /**
127    * Return a "stripped" {@link SuperCommunication} with extraneous annotations removed.
128    */
129   public SuperCommunication stripAnnotations() {
130     Communication copy = new Communication(this.comm);
131 
132     // Unset annotation fields if set.
133     if (copy.isSetEntitySetList())
134       copy.unsetEntitySetList();
135     if (copy.isSetEntityMentionSetList())
136       copy.unsetEntityMentionSetList();
137     if (copy.isSetSituationSetList())
138       copy.unsetSituationSetList();
139     if (copy.isSetSituationMentionSetList())
140       copy.unsetSituationMentionSetList();
141     if (copy.isSetLidList())
142       copy.unsetLidList();
143 
144     return new SuperCommunication(copy);
145   }

147   /**
148    * Take in a {@link Path} to an output file, and whether or not to delete the file at that path if it already exists, and output a byte array that represents
149    * a serialized {@link Communication} object.
150    *
151    * @param path
152    *          - a {@link Path} to the destination of the serialized {@link Communication}.
153    * @param deleteExisting
154    *          - whether to delete the file at path, if it exists.
155    * @throws ConcreteException
156    *           if there are {@link IOException}s or {@link TException}s.
157    */
158   public void writeToFile(Path path, boolean deleteExisting) throws ConcreteException {
159     try {
160       if (deleteExisting)
161         Files.deleteIfExists(path);
162       else if (Files.exists(path))
163         throw new ConcreteException("File exists at: " + path.toString() + ". Delete it, or " + "call this method with the second parameter set to 'true'.");
164 
165       byte[] bytez = this.ser.toBytes(this.comm);
166       Files.write(path, bytez);
167     } catch (IOException e) {
168       throw new ConcreteException(e);
169     }
170   }

188    * Get the first {@link Section} from the wrapped {@link Communication}.
189    *
190    * @return the first {@link Section}
191    * @throws ConcreteException
192    *           if there is no {@link Section}
193    */
194   public Section firstSection() throws ConcreteException {
195     if (this.hasSections())
196       return this.comm.getSectionList().get(0);
197     else
198       throw new ConcreteException("No such section exists.");
199   }

201   /**
202    * Get the first {@link Sentence}.
203    *
204    * @return the first {@link Sentence}
205    * @throws ConcreteException
206    *           if there is no {@link Sentence}, or if any prerequisites are missing.
207    */
208   public Sentence firstSentence() throws ConcreteException {
209     if (this.hasSections()) {
210       Section s = this.firstSection();
211       if (s.isSetSentenceList() && s.getSentenceListSize() > 0)
212         return s.getSentenceListIterator().next();
213       else
214         throw new ConcreteException("This communication does not have sentences.");
215     } else {
216       throw new ConcreteException("No sections, so no sentences in this Communication.");
217     }
218   }

220   /**
221    * Get the first {@link Tokenization}.
222    *
223    * @return the first {@link Tokenization}
224    * @throws ConcreteException
225    *           if there is no {@link Tokenization}, or if any of the prerequisites are missing.
226    */
227   public Tokenization firstTokenization() throws ConcreteException {
228     if (this.hasSections()) {
229       Section s = this.firstSection();
230       if (s.isSetSentenceList() && s.getSentenceListSize() > 0) {
231         Sentence firstSent =  s.getSentenceListIterator().next();
232         if (firstSent.isSetTokenization())
233           return firstSent.getTokenization();
234         else
235           throw new ConcreteException("The first sentence does not have a tokenization.");
236       } else
237         throw new ConcreteException("This communication does not have sentences in the first section.");
238     } else {
239       throw new ConcreteException("No sections, so no tokenizations in this Communication.");
240     }
241   }

244    * @return true if {@link Section}(s) are present
245    */
246   public boolean hasSections() {
247     return this.comm.isSetSectionList() && this.comm.getSectionListSize() > 0;
248   }

298       this.generateTokenizationIdToTokenizationMap();

308   private final Map<UUID, Tokenization> tokenizationIdToTokenizationMap() {
309     final Map<UUID, Tokenization> toRet = new LinkedHashMap<>();
310     List<Sentence> stList = new ArrayList<>(this.sentIdToSentenceMap.values());
311     for (Sentence st : stList) {
312       Tokenization tok = st.getTokenization();
313       UUID tId = tok.getUuid();
314       toRet.put(tId, tok);
315     }
316 
317     return toRet;
318   }

320   /**
321    * Returns a nested map. <br>
322    * <br>
323    * Top level: <br>
324    * Key: Tokenization ID <br>
325    * Value: Map[Integer, Token] that represents [ID, Token] for this {@link Tokenization} <br>
326    * <br>
327    * Nested Map: <br>
328    * Key: Token Sequence ID <br>
329    * Value: {@link Token} object
330    *
331    */
332   private Map<UUID, Map<Integer, Token>> tokenizationIdToTokenSeqIdToTokensMap() {
333     Map<UUID, Map<Integer, Token>> toRet = new LinkedHashMap<>();
334     for (Tokenization t : this.tokenizationIdToTokenizationMap.values()) {
335       UUID tId = t.getUuid();
336       Map<Integer, Token> idToTokenMap = new LinkedHashMap<Integer, Token>();
337       if (t.isSetTokenList())
338         for (Token tok : t.getTokenList().getTokenList()) {
339           idToTokenMap.put(tok.getTokenIndex(), tok);
340           toRet.put(tId, idToTokenMap);
341         }
342     }
343 
344     return toRet;
345   }

347   /* (non-Javadoc)
348    * @see concrete.util.ConcreteSentenced#generateSentenceIdToSectionMap()
349    */
350   @Override
351   public Map<UUID, Sentence> generateSentenceIdToSentenceMap() {
352     // if the section cache is not set up, do it first.
353     if (this.sectionIdToSectionMap == null)
354       this.generateSectionIdToSectionMap();
355 
356     // if run before, return.
357     if (this.sentIdToSentenceMap != null)
358       return new LinkedHashMap<>(this.sentIdToSentenceMap);
359     else {
360       final Map<UUID, Sentence> toRet = new LinkedHashMap<>();
361 
362       List<Section> sectList = new ArrayList<>(this.sectionIdToSectionMap.values());
363       for (Section s : sectList)
364         if (s.isSetSentenceList())
365           for (Sentence st : s.getSentenceList())
366             toRet.put(st.getUuid(), st);
367 
368       this.sentIdToSentenceMap = toRet;
369       return new LinkedHashMap<>(toRet);
370     }
371   }
