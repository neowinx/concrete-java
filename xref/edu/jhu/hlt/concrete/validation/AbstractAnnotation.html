6   package edu.jhu.hlt.concrete.validation;
7   
8   import java.util.UUID;
9   
10  import org.apache.thrift.TBase;
11  import org.apache.thrift.TFieldIdEnum;
12  import org.slf4j.Logger;
13  import org.slf4j.LoggerFactory;
14  
15  import edu.jhu.hlt.concrete.Communication;
16  import edu.jhu.hlt.concrete.uuid.UUIDFactory;
17  
18  /**
19   * Abstract validation class. Focusing on immutable values, attempt to
20   * easily provide validation methods that can be used in downstream tools.
21   *
22   * @author max
23   */
24  public abstract class AbstractAnnotation<T extends TBase<T, ? extends TFieldIdEnum>> {
25  
26    protected final Logger logger = LoggerFactory.getLogger(AbstractAnnotation.class);
27    protected final T annotation;
28  
29    /**
30     *
31     */
32    /*
33     * I don't think this will ever throw, but the possibility exists.
34     */
35    @SuppressWarnings("unchecked")
36    public AbstractAnnotation(T annotation) {
37      this.annotation = (T) annotation.deepCopy();
38    }
39  
40    /**
41     * Implementors should implement this method, which will be called to
42     * attempt to validate the annotation object, via {@link #validate(Communication)}.
43     * <br>
44     * <br>
45     * As a result, you don't need to call {@link #isValid()} on everything
46     * in your implementation of {@link #isValidWithComm(Communication)}.
47     *
48     * @see #validate(Communication)
49     * @param c - A communication associated with the {@link AbstractAnnotation}.
50     * @return <code>true</code> if valid.
51     */
52    protected abstract boolean isValidWithComm(Communication c);
53  
54    /**
55     * Annotations also have a notion of 'validity' on their own. For example,
56     * a {@link UUID} string may not be a correct UUID string, thus invalidating
57     * the individual annotation.
58     *
59     * @return <code>true</code> if the individual annotation is valid.
60     */
61    public abstract boolean isValid();
62  
63    /**
64     * Public validation interface.
65     * <br>
66     * <br>
67     * Note that the call to {@link #isValid()} will short-circuit,
68     * avoiding unnecessary further validation if <code>false</code>.
69     *
70     * @param c a {@link Communication} to validate
71     * @return <code>true</code> if both the individual annotation is valid
72     * and it is valid in context of the passed in {@link Communication} object.
73     */
74    public boolean validate(Communication c) {
75      return this.isValid() && this.isValidWithComm(c);
76    }
77  
78    /**
79     * Get a copy of the annotation.
80     * <br>
81     * <br>
82     * For sanity's sake, return a copy, not the original. This prevents issues
83     * when fields are mutated after the {@link AbstractAnnotation} is created.
84     */
85    /*
86     * I don't think this will ever throw, but the possibility exists.
87     */
88    @SuppressWarnings("unchecked")
89    public T getAnnotation() {
90      return (T) this.annotation.deepCopy();
91    }
92  
93    /**
94     * Print a message, then evaluate a boolean condition.
95     * <br>
96     * <br>
97     * This stinks, but easiest way I could see about informing the user
98     * about invalid things.
99     *
100    * @param st - the string to print
101    * @param exp - the returned boolean value
102    * @return 2nd parameter (boolean)
103    */
104   protected boolean printStatus(String st, boolean exp) {
105     logger.trace("Evaluating annotation: {}", this.annotation.toString());
106     String paramd = "\t{} = {}";
107     if (exp)
108       logger.trace(paramd, st, "OK");
109     else
110       logger.info(paramd, st, "Failed");
111 
112     return exp;
113   }
114 
115   /**
116    * Convenience method for UUID checking.
117    *
118    * @param uuidToCheck
119    * @return true if it is a valid UUID
120    */
121   protected boolean validateUUID(String uuidToCheck) {
122     return this.printStatus("UUID string must be valid UUID", UUIDFactory.isValidUUID(uuidToCheck));
123   }
124 
125   protected boolean validateUUID(edu.jhu.hlt.concrete.UUID uuid) {
126     if (uuid != null)
127       return this.validateUUID(uuid.getUuidString());
128     else
129       return this.printStatus("UUID was null!", false);
130   }
131 }
