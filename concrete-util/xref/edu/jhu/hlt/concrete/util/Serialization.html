5   package edu.jhu.hlt.concrete.util;
6   
7   import java.io.IOException;
8   import java.nio.file.Files;
9   import java.nio.file.Path;
10  import java.nio.file.Paths;
11  
12  import org.apache.thrift.TBase;
13  import org.apache.thrift.TDeserializer;
14  import org.apache.thrift.TException;
15  import org.apache.thrift.TFieldIdEnum;
16  import org.apache.thrift.TSerializer;
17  import org.apache.thrift.protocol.TBinaryProtocol;
18  import org.apache.thrift.protocol.TProtocolFactory;
19  
20  import edu.jhu.hlt.concrete.Communication;
21  import edu.jhu.hlt.concrete.serialization.ThriftSerializer;
22  
23  /**
24   * Will be removed in a future release.
25   *
26   * @deprecated
27   * @see ThriftSerializer
28   */
29  @Deprecated
30  public class Serialization {
31  
32    private final TProtocolFactory strategy;
33  
34    /**
35     * Use a default strategy, {@link TBinaryProtocol}, to de/serialize {@link Communication} objects.
36     */
37    public Serialization() {
38      this.strategy = new TBinaryProtocol.Factory();
39    }
40  
41    /**
42     * Generic method to serialize a thrift-like object.
43     *
44     * @param object - a 'thrift-like' {@link TBase}] object that can be used by
45     * {@link TSerializer#serialize(TBase)} to produce a byte array.
46     * @throws ConcreteException
47     */
48    public <T extends TBase<T, ? extends TFieldIdEnum>> byte[] toBytes(T object) throws ConcreteException {
49      try {
50        return new TSerializer(this.strategy).serialize(object);
51      } catch (TException e) {
52        throw new ConcreteException("Error during serialization.", e);
53      }
54    }
55  
56    /**
57     * Generic method to deserialize a thrift-like object.
58     *
59     * @param object - a 'thrift-like' [{@link TBase}] object that will be deserialized into. In other words,
60     * if you were reading in a {@link Communication} byte array, you should pass in a <code>new Communication()</code>
61     * object as the first parameter.
62     * @param bytez - the byte array that holds the serialized {@link TBase} object.
63     * @return a deserialized {@link TBase} object.
64     * @throws ConcreteException if there is an error during deserialization.
65     */
66    public <T extends TBase<T, ? extends TFieldIdEnum>> T fromBytes(T object, byte[] bytez) throws ConcreteException {
67      try {
68        new TDeserializer(this.strategy).deserialize(object, bytez);
69        return object;
70      } catch (TException e) {
71        throw new ConcreteException("Error during deserialization.", e);
72      }
73    }
74  
75    /**
76     * Same as {@link #fromBytes(TBase, byte[])}, but takes in a {@link Path} object.
77     *
78     * @see #fromBytes(TBase, byte[])
79     */
80    public <T extends TBase<T, ? extends TFieldIdEnum>> T fromBytes(T object, Path pathToSerializedFile) throws ConcreteException {
81      try {
82        return this.fromBytes(object, Files.readAllBytes(pathToSerializedFile));
83      } catch (IOException e) {
84        throw new ConcreteException(e);
85      }
86    }
87  
88    /**
89     * Same as {@link #fromBytes(TBase, byte[])}, but takes in a {@link String} that represents
90     * a path to a serialized {@link TBase} object on disk.
91     *
92     * @see #fromBytes(TBase, Path)
93     */
94    public <T extends TBase<T, ? extends TFieldIdEnum>> T fromBytes(T object, String pathToSerializedFileString) throws ConcreteException {
95      return this.fromBytes(object, Paths.get(pathToSerializedFileString));
96    }
97  }
