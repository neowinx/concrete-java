6   package edu.jhu.hlt.concrete.communications;
7   
8   import edu.jhu.hlt.concrete.Communication;
9   import edu.jhu.hlt.concrete.Section;
10  import edu.jhu.hlt.concrete.UUID;
11  import edu.jhu.hlt.concrete.section.SingleSectionSegmenter;
12  import edu.jhu.hlt.concrete.util.ConcreteException;
13  import edu.jhu.hlt.concrete.uuid.UUIDFactory;
14  
15  /**
16   * Class that allows for construction of semi-built {@link Communication}
17   * objects.
18   */
19  public class CommunicationFactory {
20  
21    private CommunicationFactory() {
22  
23    }
24  
25    /**
26     * @return a {@link Communication} with a {@link UUID} assigned.
27     */
28    public static final Communication create() {
29      return new Communication()
30        .setUuid(UUIDFactory.newUUID());
31    }
32  
33    /**
34     * @param id the ID of the newly created {@link Communication} object. Must
35     * be non-null and non-empty.
36     * @return a {@link Communication} with both a {@link UUID} and an ID.
37     * @throws ConcreteException if any input strings are null or length 0.
38     */
39    public static final Communication create(String id) throws ConcreteException {
40      if (id == null || id.length() == 0)
41        throw new ConcreteException("ID was null or length 0.");
42      return create()
43        .setId(id);
44    }
45  
46    /**
47     *
48     * @param id the ID of the newly created {@link Communication} object. Must
49     * be non-null and non-empty.
50     * @param text the text of the new Communication. Must be non-null and non-empty.
51     * @return a {@link Communication} with a {@link java.util.UUID}, ID, and text.
52     * @throws ConcreteException ConcreteException if any input strings are null or length 0.
53     */
54    public static final Communication create(String id, String text) throws ConcreteException {
55      if (text == null || text.length() == 0)
56        throw new ConcreteException("Text was null or length 0.");
57      return create(id)
58        .setText(text);
59    }
60  
61    /**
62     * @param id the ID of the new Communication. Must be non-null and non-empty.
63     * @param text the text of the new Communication. Must be non-null and non-empty.
64     * @param sectionKind the sectionKind of the new Communication. Must be non-null and non-empty.
65     * @return a {@link Communication} with a {@link java.util.UUID}, ID, text, and single {@link Section}
66     * that covers the whole text.
67     * @throws ConcreteException if any input strings are null or length 0.
68     */
69    public static final Communication create(String id, String text, String sectionKind) throws ConcreteException {
70      if (sectionKind == null || sectionKind.length() == 0)
71        throw new ConcreteException("sectionKind was null or length 0. Use 'other' if unsure.");
72      Communication c = create(id, text);
73      c.addToSectionList(SingleSectionSegmenter.createSingleSection(text, sectionKind));
74      return c;
75    }
76  }
