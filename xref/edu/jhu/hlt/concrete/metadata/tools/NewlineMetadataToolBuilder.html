6   package edu.jhu.hlt.concrete.metadata.tools;
7   
8   import edu.jhu.hlt.concrete.AnnotationMetadata;
9   
10  /**
11   * Class that builds {@link MetadataTool} objects, split by newline.
12   * Useful for creating parseable {@link AnnotationMetadata} tool names.
13   */
14  public class NewlineMetadataToolBuilder {
15  
16    private static final String NIX_NEWLINE = "\n";
17  
18    /**
19     * @param tool the {@link MetadataTool} used to build the String
20     * @return a String suitable for use in {@link AnnotationMetadata#setTool(String)}.
21     */
22    public static final String generateToolString(MetadataTool tool) {
23      final StringBuilder sb = new StringBuilder();
24      sb.append(tool.getToolName());
25      sb.append(NIX_NEWLINE);
26      sb.append(tool.getToolVersion());
27      sb.append(NIX_NEWLINE);
28      tool.getToolNotes().forEach(s -> {
29        sb.append(s);
30        sb.append(NIX_NEWLINE);
31      });
32  
33      return sb.toString();
34    }
35  }
