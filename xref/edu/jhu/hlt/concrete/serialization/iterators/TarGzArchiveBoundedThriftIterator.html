6   package edu.jhu.hlt.concrete.serialization.iterators;
7   
8   import java.io.IOException;
9   import java.io.InputStream;
10  
11  import org.apache.thrift.TBase;
12  import org.apache.thrift.TFieldIdEnum;
13  
14  import edu.jhu.hlt.acute.iterators.tar.TarGzArchiveEntryByteIterator;
15  import edu.jhu.hlt.concrete.serialization.BoundedThriftSerializer;
16  import edu.jhu.hlt.concrete.util.ConcreteException;
17  import edu.jhu.hlt.utilt.AutoCloseableIterator;
18  
19  /**
20   *
21   */
22  public class TarGzArchiveBoundedThriftIterator<T extends TBase<T, ? extends TFieldIdEnum>> implements AutoCloseableIterator<T> {
23  
24    private final BoundedThriftSerializer<T> ser;
25    private final TarGzArchiveEntryByteIterator byteIter;
26  
27    /**
28     * @throws IOException
29     * @throws ConcreteException
30     */
31    public TarGzArchiveBoundedThriftIterator(Class<T> clazz, InputStream is) throws IOException, ConcreteException {
32      this.byteIter = new TarGzArchiveEntryByteIterator(is);
33      this.ser = new BoundedThriftSerializer<>(clazz);
34    }
35  
36    @Override
37    public void close() throws IOException {
38      this.byteIter.close();
39    }
40  
41    @Override
42    public boolean hasNext() {
43      return this.byteIter.hasNext();
44    }
45  
46    @Override
47    public T next() {
48      // Throw earlier for clarity.
49      byte[] n = this.byteIter.next();
50      try {
51        // Below can throw if T is not accurately mapped to the inputStream.
52        // e.g., if it is an archive of Sections, and T is Sentences.
53        return this.ser.fromBytes(n);
54      } catch (ConcreteException e) {
55        throw new IllegalArgumentException("Exception occurred during conversion to deserialized"
56            + " form. Ensure the InputStream represents the same type as T.", e);
57      }
58    }
59  
60    @Override
61    public void remove() {
62      // throws UnsupportedOperationException
63      this.byteIter.remove();
64    }
65  }
