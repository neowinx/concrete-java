18  import org.apache.commons.compress.archivers.tar.TarArchiveEntry;
19  import org.apache.commons.compress.archivers.tar.TarArchiveOutputStream;
20  import org.apache.commons.compress.compressors.gzip.GzipCompressorOutputStream;

23  import edu.jhu.hlt.concrete.Communication;

28   * @author max

31  public class TarGzCompactCommunicationSerializer extends TarCompactCommunicationSerializer implements CommunicationTarGzSerializer {
32  
33    @Override
34    public Iterator<Communication> fromTarGz(InputStream is) throws ConcreteException, IOException {
35      return new TarGzArchiveEntryCommunicationIterator(is);
36    }
37    
38    @Override
39    public void toTarGz(Collection<Communication> commColl, Path outPath) throws ConcreteException {
40      try(OutputStream os = Files.newOutputStream(outPath);
41          BufferedOutputStream bos = new BufferedOutputStream(os);
42          GzipCompressorOutputStream gzos = new GzipCompressorOutputStream(bos);
43          TarArchiveOutputStream tos = new TarArchiveOutputStream(gzos);) {
44        for (Communication c : commColl) {
45          TarArchiveEntry entry = new TarArchiveEntry(c.getId() + ".concrete");
46          byte[] cbytes = this.toBytes(c);
47          entry.setSize(cbytes.length);
48          tos.putArchiveEntry(entry);
49          try (ByteArrayInputStream bis = new ByteArrayInputStream(cbytes)) {
50            IOUtils.copy(bis, tos);
51            tos.closeArchiveEntry();
52          }
53        }
54        
55      } catch (IOException e) {
56        throw new ConcreteException(e);
57      }
58    }
59  
60    @Override
61    public void toTarGz(Collection<Communication> commColl, String outPathString) throws ConcreteException {
62      this.toTarGz(commColl, Paths.get(outPathString));
63    }
64  }
