5   package edu.jhu.hlt.concrete.communications;
6   
7   import edu.jhu.hlt.concrete.AnnotationMetadata;
8   import edu.jhu.hlt.concrete.Communication;
9   import edu.jhu.hlt.concrete.metadata.tools.TooledMetadataConverter;
10  import edu.jhu.hlt.concrete.safe.communications.SafeCommunication;
11  import edu.jhu.hlt.concrete.safe.metadata.SafeAnnotationMetadata;
12  import edu.jhu.hlt.concrete.uuid.UUIDFactory;
13  
14  /**
15   * Class that converts {@link SafeCommunication} objects to Concrete {@link Communication}
16   * objects, which can then be added to.
17   */
18  public class SafeCommunicationConverter {
19    /**
20     * @param sc a {@link SafeCommunication}
21     * @return a {@link Communication} with required fields filled in
22     */
23    public static Communication toCommunication(SafeCommunication sc) {
24      SafeAnnotationMetadata md = sc.getMetadata();
25      AnnotationMetadata cmd = TooledMetadataConverter.convert(md);
26      
27      Communication cc = new Communication()
28          .setId(sc.getId())
29          .setUuid(UUIDFactory.fromJavaUUID(sc.getUuid()))
30          .setType(sc.getType())
31          .setMetadata(cmd);
32      return cc;
33    }
34  }
