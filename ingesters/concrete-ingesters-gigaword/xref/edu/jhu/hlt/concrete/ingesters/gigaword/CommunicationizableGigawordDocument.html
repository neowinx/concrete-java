108   /*
109    * (non-Javadoc)
110    * 
111    * @see edu.jhu.hlt.concrete.ingesters.base.communications.Communicationizable#toCommunication()
112    */
113   @Override
114   public Communication toCommunication() throws ConcreteException {
115     Communication c = new Communication()
116         .setUuid(UUIDFactory.newUUID())
117         .setId(gd.getId())
118         .setStartTime(gd.getMillis() / 1000)
119         .setType(gd.getType().toString())
120         .setText(gd.getText())
121         .setMetadata(TooledMetadataConverter.convert(this));
122 
123     List<Section> sectList = new ArrayList<Section>();
124     int nCtr = 0;
125     for (TextSpan ts : gd.getTextSpans()) {
126       Section s = new Section().setUuid(UUIDFactory.newUUID())
127           .setKind("Passage")
128           .setTextSpan(new edu.jhu.hlt.concrete.TextSpan(ts.getStart(), ts.getEnding()));
129       s.addToNumberList(nCtr);
130       nCtr += 1;
131       sectList.add(s);
132     }
133 
134     boolean hasHeadline = gd.getHeadline().isPresent();
135     boolean hasDateline = gd.getDateline().isPresent();
136 
137     // Headline + dateline --> Section 1 == Title, Section 2 == Dateline
138     if (hasHeadline && hasDateline) {
139       sectList.get(0).setKind("Title");
140       sectList.get(1).setKind("Dateline");
141       // Only headline --> Section 1 == Title
142     } else if (hasHeadline && !hasDateline)
143       sectList.get(0).setKind("Title");
144     // Only dateline --> Section 1 == Dateline
145     else if (!hasHeadline && hasDateline)
146       sectList.get(0).setKind("Dateline");
147 
148     c.setSectionList(sectList);
149     return c;
150   }

166     notes.add("If the document contains a title and a headline, the second section will be the headline.");
167     notes.add("If the document contains no title and a headline, the first section will be the headline.");
168     return notes;
169   }
