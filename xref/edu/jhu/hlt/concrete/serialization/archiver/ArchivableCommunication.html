5   package edu.jhu.hlt.concrete.serialization.archiver;
6   
7   import edu.jhu.hlt.acute.archivers.Archivable;
8   import edu.jhu.hlt.concrete.Communication;
9   import edu.jhu.hlt.concrete.serialization.CompactCommunicationSerializer;
10  import edu.jhu.hlt.concrete.util.ConcreteException;
11  
12  /**
13   * Wrapper around {@link Communication} that implements {@link Archivable}.
14   */
15  public class ArchivableCommunication implements Archivable {
16  
17    private final Communication comm;
18    // Cache id and serializer.
19    private final String id;
20    private final CompactCommunicationSerializer cs;
21  
22    private static final String ext = ".concrete";
23  
24    /**
25     * Wrap a {@link Communication} object.
26     */
27    public ArchivableCommunication(Communication comm) {
28      this.comm = comm;
29      this.id = comm.getId();
30      this.cs = new CompactCommunicationSerializer();
31    }
32  
33    /* (non-Javadoc)
34     * @see edu.jhu.hlt.acute.archivers.Archivable#getFileName()
35     */
36    @Override
37    public String getFileName() {
38      return this.id + ext;
39    }
40  
41    /* (non-Javadoc)
42     * @see edu.jhu.hlt.acute.archivers.Archivable#getBytes()
43     */
44    @Override
45    public byte[] getBytes() {
46      try {
47        return this.cs.toBytes(this.comm);
48      } catch (ConcreteException e) {
49        throw new RuntimeException(e);
50      }
51    }
52  }
