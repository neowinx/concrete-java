5   package edu.jhu.hlt.concrete.uuid;
6   
7   import edu.jhu.hlt.concrete.UUID;
8   
9   /**
10   * A class that offers Concrete {@link UUID} utilities.
11   */
12  public class UUIDFactory {
13  
14    private UUIDFactory() {
15  
16    }
17  
18    /**
19     * @return a new Concrete {@link UUID} that wraps a Java {@link java.util.UUID}.
20     */
21    public static UUID newUUID() {
22      return new UUID(java.util.UUID.randomUUID().toString());
23    }
24    
25    /**
26     * @param uuid a {@link java.util.UUID}
27     * @return a Concrete {@link UUID}
28     */
29    public static UUID fromJavaUUID(java.util.UUID uuid) {
30      return new UUID(uuid.toString());
31    }
32  
33    /**
34     * Guess, then determine, if a string is truly a {@link UUID} string.
35     *
36     * Unfortunately, an exception will be thrown if it is not valid.
37     *
38     * @param uuidStr a {@link String} representing a UUID to check
39     * @return true if a valid {@link java.util.UUID} string, otherwise false.
40     */
41    public static boolean isValidUUID(String uuidStr) {
42      // Do what is possible to avoid an exception.
43      if (uuidStr != null && uuidStr.length() == 36 && uuidStr.contains("-"))
44        // Could do more above, but just eat the exception
45        // if it's invalid
46        try {
47          java.util.UUID.fromString(uuidStr);
48          return true;
49        } catch (IllegalArgumentException iae) {
50          return false;
51        }
52      else
53        return false;
54    }
55  }
