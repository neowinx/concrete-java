6   package edu.jhu.hlt.concrete.metadata;
7   
8   import edu.jhu.hlt.concrete.AnnotationMetadata;
9   import edu.jhu.hlt.concrete.util.Timing;
10  
11  /**
12   * Class that allows for construction of semi-built {@link AnnotationMetadata}
13   * objects.
14   */
15  public class AnnotationMetadataFactory {
16  
17    private AnnotationMetadataFactory() {
18  
19    }
20  
21    /**
22     * @return an {@link AnnotationMetadata} object with the timestamp
23     * set to the current "Unix" time, in UTC.
24     */
25    public static final AnnotationMetadata fromCurrentUTCTime() {
26      return new AnnotationMetadata()
27        .setTimestamp(Timing.currentLocalTime());
28    }
29  
30    /**
31     * @return an {@link AnnotationMetadata} object with the timestamp
32     * set to the current "Unix" time, in the local time zone.
33     */
34    public static final AnnotationMetadata fromCurrentLocalTime() {
35      return new AnnotationMetadata()
36        .setTimestamp(Timing.currentLocalTime());
37    }
38  }
