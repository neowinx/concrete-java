6   package edu.jhu.hlt.concrete.metadata.tools;
7   
8   import edu.jhu.hlt.concrete.Digest;
9   import edu.jhu.hlt.concrete.safe.metadata.SafeAnnotationMetadata;
10  
11  /**
12   * Interface that extends the functionality of {@link SafeAnnotationMetadata},
13   * requiring implementation a {@link MetadataTool}.
14   * <br>
15   * <br>
16   * Additionally, contains a default implementation of {@link SafeAnnotationMetadata#getTool()}
17   * that uses the provided MetadataTool to create a well-formed Tool string.
18   */
19  public interface SafeTooledAnnotationMetadata extends SafeAnnotationMetadata, MetadataTool {
20  
21    @Override
22    default String getToolName() {
23      return this.getClass().getName();
24    }
25    
26    /**
27     * By default, use {@link NewlineMetadataToolBuilder#generateToolString(MetadataTool)}
28     * to generate a parseable Tool string. 
29     * <br>
30     * <br>
31     * Implementers should generally not override this implementation.
32     */
33    /*
34     * (non-Javadoc)
35     * @see edu.jhu.hlt.concrete.safe.metadata.SafeAnnotationMetadata#getTool()
36     */
37    @Override
38    default String getTool() {
39      final StringBuilder sb = new StringBuilder();
40      sb.append(this.getToolName());
41      sb.append(" ");
42      sb.append(this.getToolVersion());
43      return sb.toString();
44    }
45    
46    /**
47     * @return a {@link Digest} with the string list field populated
48     * with notes about the tool.
49     */
50    default Digest getToolNotesDigest() {
51      return new Digest().setStringList(this.getToolNotes());
52    }
53  }
