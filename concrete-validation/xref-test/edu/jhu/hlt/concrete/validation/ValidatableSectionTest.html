6   package edu.jhu.hlt.concrete.validation;
7   
8   import static org.junit.Assert.assertFalse;
9   import static org.junit.Assert.assertTrue;
10  
11  import org.junit.After;
12  import org.junit.Before;
13  import org.junit.Test;
14  
15  import edu.jhu.hlt.concrete.Communication;
16  import edu.jhu.hlt.concrete.Section;
17  import edu.jhu.hlt.concrete.TextSpan;
18  import edu.jhu.hlt.concrete.UUID;
19  import edu.jhu.hlt.concrete.communications.SuperCommunication;
20  import edu.jhu.hlt.concrete.util.ConcreteException;
21  import edu.jhu.hlt.concrete.uuid.UUIDFactory;
22  
23  /**
24   *
25   */
26  public class ValidatableSectionTest extends AbstractValidationTest {
27  
28    public Section generateSection() {
29      Section s = new Section()
30        .setUuid(UUIDFactory.newUUID())
31        .setTextSpan(new TextSpan(0, this.comm.getText().length()))
32        .setKind("Passage");
33      return s;
34    }
35  
36    public Communication constructComm() {
37      Communication cpy = new Communication(this.comm);
38      cpy.addToSectionList(this.generateSection());
39      return cpy;
40    }
41  
42    /**
43     * @throws java.lang.Exception
44     */
45    @Before
46    public void setUp() throws Exception {
47    }
48  
49    /**
50     * @throws java.lang.Exception
51     */
52    @After
53    public void tearDown() throws Exception {
54    }
55  
56    @Test
57    public void validOK() throws ConcreteException {
58      Communication v = this.constructComm();
59      ValidatableSection vs = new ValidatableSection(new SuperCommunication(v).firstSection());
60      assertTrue(vs.isValid());
61      assertTrue(vs.validate(v));
62    }
63  
64    @Test
65    public void badUuid() throws ConcreteException {
66      Communication v = this.constructComm();
67      Section s = new SuperCommunication(v).firstSection();
68      s.setUuid(new UUID("3k3pgjp3"));
69      ValidatableSection vs = new ValidatableSection(s);
70      assertFalse(vs.isValid());
71      assertFalse(vs.validate(v));
72    }
73  
74    @Test
75    public void badTextSpan() throws ConcreteException {
76      Communication v = this.constructComm();
77      Section s = new SuperCommunication(v).firstSection();
78      s.setTextSpan(new TextSpan(-1, 0));
79      ValidatableSection vs = new ValidatableSection(s);
80      assertFalse(vs.isValid());
81      assertFalse(vs.validate(v));
82    }
83  
84    @Test
85    public void noKind() throws ConcreteException {
86      Communication v = this.constructComm();
87      Section s = new SuperCommunication(v).firstSection();
88      s.unsetKind();
89      ValidatableSection vs = new ValidatableSection(s);
90      assertFalse(vs.isValid());
91      assertFalse(vs.validate(v));
92    }
93  
94    @Test
95    public void misalignedTextSpan() throws ConcreteException {
96      Communication v = this.constructComm();
97      Section s = new SuperCommunication(v).firstSection();
98      s.setTextSpan(new TextSpan(0, v.getText().length() + 5));
99      ValidatableSection vs = new ValidatableSection(s);
100     assertTrue(vs.isValid());
101     assertFalse(vs.validate(v));
102   }
103 }
