6   package edu.jhu.hlt.concrete.util;
7   
8   import java.util.HashSet;
9   import java.util.Random;
10  import java.util.Set;
11  
12  import org.slf4j.Logger;
13  import org.slf4j.LoggerFactory;
14  
15  import edu.jhu.hlt.concrete.AnnotationMetadata;
16  import edu.jhu.hlt.concrete.Communication;
17  import edu.jhu.hlt.concrete.communications.SuperCommunication;
18  import edu.jhu.hlt.concrete.random.RandomConcreteFactory;
19  
20  /**
21   * See {@link RandomConcreteFactory}.
22   *
23   * @deprecated
24   * @see RandomConcreteFactory
25   */
26  @Deprecated
27  public class ConcreteFactory {
28  
29    private final Random r;
30  
31    private static final String[] COMM_TYPES = new String[] { "Document", "Tweet", "Email" };
32    private static final int COMM_TYPE_SIZE = COMM_TYPES.length;
33    private static final Logger logger = LoggerFactory.getLogger(ConcreteFactory.class);
34  
35    /**
36     *
37     */
38    public ConcreteFactory() {
39      this.r = new Random();
40    }
41  
42    public ConcreteFactory(long seed) {
43      this.r = new Random(seed);
44    }
45  
46    public String randomCommunicationType() {
47      return COMM_TYPES[this.r.nextInt(COMM_TYPE_SIZE)];
48    }
49  
50    /**
51     * Generate a {@link Communication} object with basic fields set.
52     */
53    public Communication randomCommunication() {
54      return new Communication()
55        .setUuid(new ConcreteUUIDFactory().getConcreteUUID())
56        .setId("corpus_" + this.r.nextInt(Integer.MAX_VALUE))
57        .setText("Some sample text.")
58        .setType(this.randomCommunicationType())
59        .setMetadata(this.randomMetadata());
60    }
61  
62    public Set<Communication> randomCommunicationSet(int nMembers) {
63      Set<Communication> cSet = new HashSet<>(nMembers + 1);
64      for (int i = 0; i < nMembers; i++)
65        cSet.add(this.randomCommunication());
66      // could get some dupes
67      while (cSet.size() < nMembers)
68        cSet.add(this.randomCommunication());
69      return cSet;
70    }
71  
72    /**
73     * Generate an {@link AnnotationMetadata} object.
74     *
75     */
76    public AnnotationMetadata randomMetadata() {
77      return new AnnotationMetadata()
78        .setTimestamp(System.currentTimeMillis())
79        .setTool("ConcreteFactory");
80    }
81  
82    public static void main (String... args) throws ConcreteException {
83      if (args.length < 1 || args.length > 2) {
84        logger.info("Usage: {} <path-to-output-file>", ConcreteFactory.class.getName());
85        logger.info("Optional argument: <delete-if-exists>");
86        logger.info("e.g., my/output/folder.concrete true");
87        System.exit(1);
88      }
89  
90      boolean overWrite = true;
91      if (args.length == 2)
92        overWrite = Boolean.parseBoolean(args[1]);
93      new SuperCommunication(new ConcreteFactory().randomCommunication()).writeToFile(args[0], overWrite);
94    }
95  }
